4pub(crate) fn push_extract_contains_candidates<L, A, CF>(
5  candidates: &mut Vec<(CF::Cost, Id)>,
6  exprs: &mut ExprHashCons<L>,
7  cost_f: &mut CF,
8  extracted: &HashMap<Id, (CF::Cost, Id)>,
9  egraph: &EGraph<L, A>,
10  enode: &L,
11  mut analysis_of: impl FnMut(Id, &mut ExprHashCons<L>, &mut CF, &HashMap<Id, (CF::Cost, Id)>, &EGraph<L, A>) -> Option<(CF::Cost, Id)>,
12)
13where
14    L: Language,
15    A: Analysis<L>,
16    CF: CostFunction<L>,
17    CF::Cost: 'static + Ord,
18{
19  // TODO: could probably simplify and optimize this code
20
21  // Children that satisfy the sketch
22  let children_matching: Vec<_> = { enode
23      .children()
24      .iter()
25      .map(|&c| {
26          let data = analysis_of(c, exprs, cost_f, extracted, egraph);
27          (c, data)
28      })
29      .collect()
30  };
31
32  // Children that satisfy '?'
33  let children_any: Vec<_> = enode
34      .children()
35      .iter()
36      .map(|&c| (c, extracted[&egraph.find(c)].clone()))
37      .collect();
38
39  let mut index_based_enode = enode.clone();
40  for (index, id) in index_based_enode.children_mut().iter_mut().enumerate() {
41      *id = Id::from(index);
42  }
43
44  // If one child satisfies sketch, it's enough.
45  // Take '?' options for the other children.
46  // Accumulate all combination.
47  for (matching_index, (matching_id, matching_data)) in children_matching.iter().enumerate() {
48      if let Some(matching_data) = matching_data {
49          let mut to_selected = HashMap::default();
50
51          for (index, (id, any_data)) in children_any.iter().enumerate() {
52              let selected = if index == matching_index {
53                  matching_data
54              } else {
55                  any_data
56              };
57              to_selected.insert(Id::from(index), selected);
58          }
59
60          candidates.push((
61              cost_f.cost(&index_based_enode, |c| to_selected[&c].0.clone()),
62              exprs
63                  .add(index_based_enode.clone().map_children(|c| to_selected[&c].1)),
64          ));
65      }
66  }
67}

69pub(crate) fn extract_only_contains_candidate<L, A, CF>(
70  exprs: &mut ExprHashCons<L>,
71  cost_f: &mut CF,
72  egraph: &EGraph<L, A>,
73  enode: &L,
74  mut analysis_of: impl FnMut(Id, &mut ExprHashCons<L>, &mut CF, &EGraph<L, A>) -> Option<(CF::Cost, Id)>,
75) -> Option<(CF::Cost, Id)>
76where
77    L: Language,
78    A: Analysis<L>,
79    CF: CostFunction<L>,
80    CF::Cost: 'static + Ord,
81{
82  if enode.children().is_empty() { return None }
83
84  // Children that satisfy the sketch
85  let children_matching: HashMap<_, _> = { enode
86      .children()
87      .iter()
88      .map(|&c| {
89          let data = analysis_of(c, exprs, cost_f, egraph);
90          (c, data)
91      })
92      .collect()
93  };
94
95  let all_contains = children_matching.iter().all(|(c, data)| {
96    data.is_some()
97  });
98  if !all_contains { return None };
99
100  Some((
101      cost_f.cost(enode, |c| children_matching[&c].as_ref().unwrap().0.clone()),
102      exprs
103          .add(enode.clone().map_children(|c| children_matching[&c].as_ref().unwrap().1)),
104  ))
